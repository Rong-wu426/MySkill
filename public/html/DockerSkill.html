Docker 筆記

安裝 Docker


    # 在 Ubuntu 上安裝 Docker
    sudo apt-get update
    sudo apt-get install docker-ce docker-ce-cli containerd.io
    
    # 驗證 Docker 是否安裝成功
    sudo docker --version

Dockerfile


    # 使用官方 Ubuntu 基礎映像
    FROM ubuntu:latest
    
    # 安裝更新和軟件包
    RUN apt-get update && apt-get install -y \
    python3 \
    python3-pip
    
    # 設置工作目錄
    WORKDIR /app
    
    # 複製目前目錄內容到容器內
    COPY . /app
    
    # 安裝 Python 套件
    RUN pip3 install -r requirements.txt
    
    # 設定容器啟動時執行的命令
    CMD ["python3", "app.py"]

建構映像


    # 使用 Dockerfile 構建映像
    docker build -t my-image .
    
    # 查看建立的映像
    docker images

運行容器


    # 運行容器
    docker run -d -p 80:80 my-image
    
    # 進入正在運行的容器
    docker exec -it container_id /bin/bash

查看運作中的容器


    # 列出運行中的容器
    docker ps

停止容器


    # 停止容器
    docker stop container_id

刪除容器


    # 刪除停止的容器
    docker rm container_id

查看映像


    # 列出所有映像
    docker images

刪除映像


    # 刪除映像
    docker rmi image_id
