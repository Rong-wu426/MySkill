類別和物件


      class Person:
          def __init__(self, name, age):
              self.name = name
              self.age = age

          def greet(self):
              return f"你好, 我是 {self.name}，我 {self.age} 歲。"

      # 創建一個 Person 物件
      person = Person("Alice", 30)
      print(person.greet())

檔案操作


      # 寫入文件
      with open('file.txt', 'w') as f:
          f.write("Hello, World!")
  
      # 讀取文件
      with open('file.txt', 'r') as f:
          content = f.read()
      print(content)
  
      # 追加到文件
      with open('file.txt', 'a') as f:
          f.write("\n追加的内容")
