類別


class Person:
  def __init__(self, name, age):
    self.name = name
    self.age = age

  def greet(self):
    return f"你好, 我是 {self.name}，我 {self.age} 歲。"

# 創建一個 Person 物件
person = Person("Alice", 30)
print(person.greet())
