Seaborn 筆記

簡介

Seaborn 是一個基於 Matplotlib 的 Python資料視覺化庫，提供高階的介面來繪製統計圖表。

安裝


# 使用 pip 安裝 Seaborn
pip install seaborn

基本繪圖


import seaborn as sns
import matplotlib.pyplot as plt

# 使用 Seaborn 繪製基本圖形
data = sns.load_dataset("tips")

sns.scatterplot(x="total_bill", y="tip", data=data)
plt.xlabel('總帳單')
plt.ylabel('小費')
plt.title('總帳單 vs 小費')
plt.show()

直方圖


import seaborn as sns
import matplotlib.pyplot as plt

# 使用 Seaborn 繪製直方圖
data = sns.load_dataset("iris")

sns.histplot(data=data, x="sepal_length", hue="species", multiple="stack")
plt.xlabel('花萼長度')
plt.ylabel('頻次')
plt.title('花萼長度的直方圖')
plt.show()

散點圖


import seaborn as sns
import matplotlib.pyplot as plt

# 使用 Seaborn 繪製散點圖
data = sns.load_dataset("iris")

sns.scatterplot(x="sepal_length", y="sepal_width", hue="species", data=data)
plt.xlabel('花萼長度')
plt.ylabel('花萼寬度')
plt.title('花萼長度 vs 花萼寬度')
plt.show()

箱型圖


import seaborn as sns
import matplotlib.pyplot as plt

# 使用 Seaborn 繪製箱型圖
data = sns.load_dataset("tips")

sns.boxplot(x="day", y="total_bill", data=data)
plt.xlabel('星期')
plt.ylabel('總帳單')
plt.title('各星期的總帳單箱型圖')
plt.show()

對角線圖


import seaborn as sns
import matplotlib.pyplot as plt

# 使用 Seaborn 繪製對角線圖
data = sns.load_dataset("iris")

sns.pairplot(data, hue="species")
plt.title('對角線圖')
plt.show()

熱力圖


import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# 使用 Seaborn 繪製熱力圖
data = np.random.rand(10, 12)
sns.heatmap(data, cmap="YlGnBu", annot=True)
plt.title('熱力圖')
plt.show()

自定義


import seaborn as sns
import matplotlib.pyplot as plt

# 使用 Seaborn 繪製自訂圖形
data = sns.load_dataset("tips")

sns.scatterplot(x="total_bill", y="tip", data=data, color='orange', marker='D', s=100)
plt.xlabel('總帳單')
plt.ylabel('小費')
plt.title('自訂散點圖')
plt.grid(True)
plt.show()
