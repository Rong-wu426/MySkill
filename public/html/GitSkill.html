Git 筆記

基本指令


      # 複製倉庫
      git clone [repository-url]
      
      # 查看狀態
      git status
      
      # 新增更改
      git add [file]
      
      # 提交更改
      git commit -m "Commit message"
      
      # 查看提交歷史
      git log

初始化倉庫


      # 初始化一個新的 Git 倉庫
      git init

配置


      # 設定使用者資訊
      git config --global user.name "Your Name"
      git config --global user.email "your.email@example.com"
      
      # 查看目前配置
      git config --list

分支管理


      # 查看所有分支
      git branch
      
      # 建立新分支
      git branch [branch-name]
      
      # 切換分支
      git checkout [branch-name]
      
      # 建立並切換到新分支
      git checkout -b [branch-name]

合併分支


    # 切換到目標分支
    git checkout [target-branch]
    
    # 合併其他分支
    git merge [source-branch]

遠端倉庫


    # 新增遠端倉庫
    git remote add origin [repository-url]
    
    # 推送到遠端倉庫
    git push origin [branch-name]
    
    # 拉取遠端倉庫更改
    git pull origin [branch-name]

快取


    # 儲存目前變更到快取區
    git stash
    
    # 應用程式快取區的更改
    git stash apply
    
    # 查看快取區列表
    git stash list

回退


    # 回退到某個提交
    git reset --hard [commit-id]
    
    # 取消對文件的更改
    git checkout -- [file]

提交歷史


    # 查看簡潔的提交歷史
    git log --oneline

查看差異


    # 查看工作區和索引區的差異
    git diff
    
    # 查看兩個提交之間的差異
    git diff [commit-id1] [commit-id2]

重設變更


    # 重設文件到最新提交狀態
    git reset [file]
    
    # 重設索引區和工作區到最新提交狀態
    git reset --hard

.gitignore


    # 創建 .gitignore 文件
    # 用來排除不需要的文件和目錄
    # 例如：
    # *.log
    # node_modules/
    # .env

Git 版本


    # 查看 Git 版本
    git version

GitHub


    # 設置遠端倉庫
    git remote add origin https://github.com/username/repository.git
    
    # 重命名默認分支為 main
    git branch -M main
    
    # 推送本地更改到 GitHub
    git push -u origin main
    
    # 克隆 GitHub 倉庫到本地
    git clone https://github.com/username/repository.git
