Matplotlib 筆記

簡介

Matplotlib 是一個 Python的資料可視化庫，可以用來產生各種靜態、動態和互動式圖表。

安裝


      # 使用 pip 安裝 Matplotlib
      pip install matplotlib

基本繪圖


      import matplotlib.pyplot as plt
  
      # 繪製基本圖形
      plt.plot([1, 2, 3, 4], [10, 20, 25, 30])
      plt.xlabel('X 軸')
      plt.ylabel('Y 軸')
      plt.title('基本圖形')
      plt.show()

折線圖


      import matplotlib.pyplot as plt
  
      # 繪製折線圖
      x = [1, 2, 3, 4]
      y = [10, 20, 25, 30]
  
      plt.plot(x, y, marker='o')
      plt.xlabel('X 軸')
      plt.ylabel('Y 軸')
      plt.title('折線圖')
      plt.grid(True)
      plt.show()

散點圖


      import matplotlib.pyplot as plt
  
      # 繪製散點圖
      x = [1, 2, 3, 4]
      y = [10, 20, 15, 25]
  
      plt.scatter(x, y, color='red', marker='x')
      plt.xlabel('X 軸')
      plt.ylabel('Y 軸')
      plt.title('散點圖')
      plt.show()

直方圖


      import matplotlib.pyplot as plt
  
      # 繪製直方圖
      data = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
  
      plt.hist(data, bins=4, color='blue', edgecolor='black')
      plt.xlabel('數值')
      plt.ylabel('頻次')
      plt.title('直方圖')
      plt.show()

子圖


      import matplotlib.pyplot as plt
  
      # 建立子圖
      fig, axs = plt.subplots(2, 2, figsize=(10, 8))
  
      axs[0, 0].plot([1, 2, 3], [1, 4, 9], 'tab:blue')
      axs[0, 0].set_title('子圖 1')
  
      axs[0, 1].scatter([1, 2, 3], [9, 4, 1], color='red')
      axs[0, 1].set_title('子圖 2')
  
      axs[1, 0].bar(['A', 'B', 'C'], [10, 20, 15], color='green')
      axs[1, 0].set_title('子圖 3')
  
      axs[1, 1].hist([1, 2, 2, 3, 3, 4], bins=3, color='purple')
      axs[1, 1].set_title('子圖 4')
  
      plt.tight_layout()
      plt.show()

自定義


      import matplotlib.pyplot as plt
  
      # 繪製自訂圖形
      x = [1, 2, 3, 4]
      y = [10, 20, 15, 25]
  
      plt.plot(x, y, color='green', linestyle='--', marker='o', markersize=8)
      plt.xlabel('X 軸')
      plt.ylabel('Y 軸')
      plt.title('自定義圖形')
      plt.grid(True, linestyle='-.')
      plt.show()
