循環語句


      // for 迴圈
      for (int i = 0; i < 5; i++)
      {
          Console.WriteLine(i);
      }

      // while 循環
      int count = 0;
      while (count < 5)
      {
          Console.WriteLine(count);
          count++;
      }

類別


      public class Person
      {
          public string Name { get; set; }
          public int Age { get; set; }

          public void Introduce()
          {
              Console.WriteLine($"我的名字是 {Name}，我 {Age} 歲。");
          }
      }

      // 創建類別實例
      Person person = new Person { Name = "Alice", Age = 30 };
      person.Introduce();

異常處理


      try
      {
          int result = 10 / 0; // 這會引發除零異常
      }
      catch (DivideByZeroException ex)
      {
          Console.WriteLine("發生錯誤: " + ex.Message);
      }
      finally
      {
          Console.WriteLine("這裡的代碼無論是否發生異常都會執行。");
      }
