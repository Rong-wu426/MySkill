Vue.js 筆記

簡介

Vue.js 是一個進階的 JavaScript 框架，用於建立使用者介面和單頁應用程式。

安裝


    # 使用 npm 安裝
    npm install vue

    # 在 HTML 中使用 CDN
    <script src="https://cdnjs.cloudflare.com/ajax/libs/lightgallery/1.6.6/js/lightgallery.min.js"></script>

基本用法


    
    <div id="app">
    {{ message }}
    </div>

    
    <script>
    new Vue({
    el: '#app',
    data: {
    message: 'Hello Vue!'
    }
    });
    </script>

指令


    
    <img v-bind:src="imageSrc" alt="Vue.js">

    
    <input v-model="inputValue" placeholder="輸入內容">

    
    <p v-if="showParagraph">這是條件渲染的內容。 </p>

    
    <ul>
    <li v-for="item in items" :key="item.id">{{ item.name }}</li>
    </ul>

組件


    
    <script>
    Vue.component('my-component', {
    template: '<div>A custom component!</div>'
    });
    </script>

    
    <my-component></my-component>

狀態管理


    # 使用 Vuex 進行狀態管理
    npm install vuex

    # 建立 Vuex store
    import Vue from 'vue';
    import Vuex from 'vuex';

    Vue.use(Vuex);

    const store = new Vuex.Store({
    state: {
    count: 0
    },
    mutations: {
    increment(state) {
    state.count++;
    }
    }
    });

    new Vue({
    el: '#app',
    store,
    computed: {
    count() {
    return this.$store.state.count;
    }
    }
    });

路由


    # 使用 Vue Router 進行路由管理
    npm install vue-router

    # 設定路由
    import Vue from 'vue';
    import Router from 'vue-router';

    Vue.use(Router);

    const routes = [
    { path: '/', component: Home },
    { path: '/about', component: About }
    ];

    const router = new Router({ routes });

    new Vue({
    el: '#app',
    router
    });
