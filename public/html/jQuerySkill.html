jQuery 筆記

選擇器


    $(document).ready(function(){
    // 選擇所有的 <p> 元素
    $("p").css("color", "blue");

    // 選擇具有特定 ID 的元素
    $("#myId").css("background-color", "yellow");

    // 選擇具有特定 class 的元素
    $(".myClass").css("font-size", "20px");
    });

事件處理


    $(document).ready(function(){
    // 點擊事件
    $("#myButton").click(function(){
    alert("按鈕被點擊了！");
    });

    // 表單提交事件
    $("form").submit(function(event){
    event.preventDefault(); // 防止表單提交
    alert("表單提交了！");
    });
    });

DOM 操作


    $(document).ready(function(){
    // 新增內容
    $("#myDiv").html("<p>新內容</p>");

    // 新增元素
    $("#myList").append("<li>新項目</li>");

    // 移除元素
    $("#myElement").remove();
    });

效果


    $(document).ready(function(){
    // 顯示元素
    $("#myElement").show();

    // 隱藏元素
    $("#myElement").hide();

    // 減少透明度
    $("#myElement").fadeTo("slow", 0.5);

    // 顯示元素並新增動畫
    $("#myElement").slideDown();
    });

AJAX


    $(document).ready(function(){
    // 發送 GET 請求
    $.ajax({
    url: "https://api.example.com/data",
    method: "GET",
    success: function(data){
    console.log(data);
    },
    error: function(error){
    console.error("發生錯誤：", error);
    }
    });

    // 發送 POST 請求
    $.ajax({
    url: "https://api.example.com/submit",
    method: "POST",
    data: { key: "value" },
    success: function(response){
    alert("資料提交成功！");
    },
    error: function(error){
    console.error("發生錯誤：", error);
    }
    });
    });
